## Quick Color Picker

This is a very simple color picker tool. Upload an image and click on it to get the color at that point.

```html
<!-- Simple image color picker -->
<div>
  <input type="file" id="imageLoader" name="imageLoader" accept="image/*" />
  <canvas id="imageCanvas" style="cursor: crosshair; display: block; margin-top: 10px;"></canvas>
  <p>Picked Color: <span id="colorValue">#ffffff</span></p>
</div>
<script>
  const imageLoader = document.getElementById('imageLoader');
  const canvas = document.getElementById('imageCanvas');
  const ctx = canvas.getContext('2d');
  const colorValue = document.getElementById('colorValue');

  imageLoader.addEventListener('change', handleImage, false);
  canvas.addEventListener('click', pickColor, false);

  function handleImage(e) {
    const reader = new FileReader();
    reader.onload = function(event) {
      const img = new Image();
      img.onload = function() {
        canvas.width = img.width;
        canvas.height = img.height;
        ctx.drawImage(img, 0, 0);
      };
      img.src = event.target.result;
    };
    reader.readAsDataURL(e.target.files[0]);
  }

  function pickColor(e) {
    const x = e.offsetX;
    const y = e.offsetY;
    const pixel = ctx.getImageData(x, y, 1, 1).data;
    const hex = rgbToHex(pixel[0], pixel[1], pixel[2]);
    colorValue.textContent = hex;
    colorValue.style.background = hex;
  }

  function rgbToHex(r, g, b) {
    return '#' + [r, g, b].map(c => c.toString(16).padStart(2, '0')).join('');
  }
</script>
```
